## Properties

<PropsTable>
  <Prop
    name="totalPages"
    type="number"
    default="—"
    description="Total number of pages. Required."
  />
  <Prop
    name="currentPage"
    type="number"
    default="—"
    description="Current active page (1-indexed). Required."
  />
  <Prop
    name="onPageChange"
    type="(page: number) => void"
    default="—"
    description="Callback when page changes. Required."
  />
  <Prop
    name="siblingCount"
    type="number"
    default="1"
    description="Number of sibling pages to show around current page."
  />
  <Prop
    name="color"
    type="PaginationColor"
    default='"default"'
    description="Color variant for the active page. Options: default, blue, purple, pink, red, orange, green, teal, cyan, indigo, violet, rose, amber, lime, sky, emerald, fuchsia."
  />
  <Prop
    name="variant"
    type="PaginationVariant"
    default='"solid"'
    description="Visual style variant. Options: solid, outline, ghost, squircle."
  />
  <Prop
    name="showEdges"
    type="boolean"
    default="false"
    description="Show first/last page buttons with double chevrons."
  />
  <Prop
    name="size"
    type='"sm" | "md" | "lg"'
    default='"md"'
    description="Size of pagination buttons."
  />
  <Prop
    name="className"
    type="string"
    default="—"
    description="Additional CSS classes for the pagination container."
  />
</PropsTable>

## Hook

### usePaginationState

A convenience hook for pagination state management.

```tsx
const { page, setPage, nextPage, prevPage, resetPage } = usePaginationState(1);
```

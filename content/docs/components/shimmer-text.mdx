---
title: Shimmer Text
description: A premium text component with a continuous shimmer effect flowing within the text itself.
category: Components
---

The ShimmerText component creates a smooth, continuous shimmer effect that flows within the text characters themselves. The shimmer loops infinitely, creating an elegant effect perfect for headings and taglines.

## Preview

<ComponentPreview name="shimmer-text">
```tsx
import { ShimmerText } from "@/components/shimmer-text";

export default function Example() {
  return (
    <ShimmerText shimmerColor="#60a5fa" className="text-4xl font-bold">
      Build beautiful interfaces
    </ShimmerText>
  );
}
```
</ComponentPreview>

## Usage

### Installation

<InstallBlock command="shimmer-text" />

### Basic Example

```tsx
import { ShimmerText } from "@/components/shimmer-text";

export default function App() {
  return (
    <ShimmerText>
      Shimmer Effect
    </ShimmerText>
  );
}
```

### Custom Shimmer Color

```tsx
import { ShimmerText } from "@/components/shimmer-text";

export default function App() {
  return (
    <ShimmerText shimmerColor="#60a5fa" className="text-4xl font-bold">
      Blue Shimmer
    </ShimmerText>
  );
}
```

### Right to Left Direction

```tsx
import { ShimmerText } from "@/components/shimmer-text";

export default function App() {
  return (
    <ShimmerText direction="rtl" shimmerColor="#a855f7" className="text-4xl font-bold">
      Reverse Direction
    </ShimmerText>
  );
}
```

### Slower Animation

```tsx
import { ShimmerText } from "@/components/shimmer-text";

export default function App() {
  return (
    <ShimmerText duration={3} shimmerColor="#60a5fa" className="text-4xl font-bold">
      Slow Shimmer
    </ShimmerText>
  );
}
```

### Large Heading

```tsx
import { ShimmerText } from "@/components/shimmer-text";

export default function App() {
  return (
    <ShimmerText 
      shimmerColor="#a855f7" 
      className="text-6xl font-bold"
    >
      Big Shimmer
    </ShimmerText>
  );
}
```

## Properties

<PropsTable>
  <Prop
    name="children"
    type="React.ReactNode"
    default="—"
    description="The text content to display with shimmer effect."
  />
  <Prop
    name="className"
    type="string"
    default="—"
    description="Additional CSS classes for styling (text size, weight, etc.)."
  />
  <Prop
    name="shimmerColor"
    type="string"
    default='"#60a5fa"'
    description="The color of the shimmer highlight that flows through the text."
  />
  <Prop
    name="direction"
    type='"ltr" | "rtl"'
    default='"ltr"'
    description="Direction of shimmer animation (left-to-right or right-to-left)."
  />
  <Prop
    name="duration"
    type="number"
    default="2"
    description="Duration of one complete shimmer cycle in seconds."
  />
  <Prop
    name="delay"
    type="number"
    default="0"
    description="Delay before the animation starts in seconds."
  />
  <Prop
    name="shimmerWidth"
    type="number"
    default="0.5"
    description="Width of the shimmer band (0.1 - 1). Higher values create wider shimmer."
  />
</PropsTable>
